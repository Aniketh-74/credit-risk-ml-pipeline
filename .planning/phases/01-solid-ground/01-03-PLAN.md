```bash
# .env.example — copy to .env and fill in your values
# DO NOT commit .env — it contains secrets

# ── PostgreSQL ────────────────────────────────────────────────────────────────
POSTGRES_USER=admin
POSTGRES_PASSWORD=<your-secure-password>
POSTGRES_DB=postgres
POSTGRES_MULTIPLE_DATABASES=airflow_db,mlflow_db,app_db

# ── Application DB (FastAPI + Alembic) ───────────────────────────────────────
# Use asyncpg driver for FastAPI; Alembic env.py will swap to psycopg2 automatically
APP_DB_URL=postgresql+asyncpg://admin:<your-secure-password>@localhost:5432/app_db

# ── MLflow ───────────────────────────────────────────────────────────────────
MLFLOW_TRACKING_URI=http://localhost:5000
MLFLOW_BACKEND_STORE_URI=postgresql+psycopg2://admin:<your-secure-password>@postgres:5432/mlflow_db

AIRFLOW__CORE__EXECUTOR=LocalExecutor
```
